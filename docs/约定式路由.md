# 约定式路由

```bash
docs
├── foo
│   └── bar.md
└── foo.md
```

```ts
import { defineConfig } from 'config'

export default defineConfig({
  route: {
    // These files will be excluded from the routing (support glob pattern)
    exclude: ['bar.md', 'foo'],
    // These files will be included in the routing (support glob pattern)
    include: ['foo'],
  },
})
```
